# 서보 제어 프로그래밍 (★★★)

#arduino #motor-servo #control

## Overview Table

| 방법 | 설명 | 권장 상황 |
|------|------|-----------|
| `delay()` 방식 | 간단하지만 blocking | 서보 1개, 학습용 |
| `millis()` 방식 | 논블로킹, 다중 서보 가능 | **프로젝트 권장** |
| PCA9685 방식 | I2C 확장, 16채널 | 서보 5개 이상 |

## 기본 서보 제어

```cpp
#include <Servo.h>

Servo myServo;

void setup() {
  myServo.attach(9);     // 핀 9에 서보 연결
}

// ⚠️ 단일 서보 테스트 전용 — 실제 작품에는 millis() 기반 논블로킹 방식 사용
void loop() {
  myServo.write(0);      // 0도로 이동
  delay(1000);
  myServo.write(90);     // 90도로 이동
  delay(1000);
  myServo.write(180);    // 180도로 이동
  delay(1000);
}
```

## 논블로킹 다중 서보 제어 (millis 방식)

> [!important] 갤럽 제어 핵심 코드 패턴
>
> ```cpp
> #include <Servo.h>
>
> Servo leg1, leg2, leg3, leg4;
> Servo rider;
>
> // 각 서보의 현재 상태와 타이밍
> unsigned long legTimer[4] = {0, 0, 0, 0};
> int legPhase[4] = {0, 0, 0, 0};  // 현재 위상
>
> // 갤럽 시퀀스 설정 (ms)
> const int LIFT_TIME    = 150;  // 다리 드는 시간
> const int HOLD_TIME    = 100;  // 공중 유지 시간
> const int LAND_TIME    = 150;  // 착지 시간
> const int GROUND_TIME  = 600;  // 착지 유지 시간
>
> // 다리 위상 오프셋 (ms)
> const int PHASE_OFFSET[4] = {0, 250, 500, 750}; // 4분의 1 주기씩
>
> void setup() {
>   leg1.attach(3);  leg2.attach(5);
>   leg3.attach(6);  leg4.attach(9);
>   rider.attach(10);
>
>   // 초기 위상 오프셋 적용
>   for (int i = 0; i < 4; i++) {
>     legTimer[i] = millis() + PHASE_OFFSET[i];
>   }
> }
>
> void updateLeg(Servo &leg, unsigned long &timer, int &phase) {
>   unsigned long now = millis();
>   if (now < timer) return; // 아직 오프셋 대기 (timer가 미래값이면 건너뜀)
>
>   unsigned long elapsed = now - timer;
>
>   if (phase == 0 && elapsed >= LIFT_TIME) {
>     leg.write(45);   // 다리 들기
>     timer = now; phase = 1;
>   } else if (phase == 1 && elapsed >= HOLD_TIME) {
>     timer = now; phase = 2;
>   } else if (phase == 2 && elapsed >= LAND_TIME) {
>     leg.write(0);    // 착지
>     timer = now; phase = 3;
>   } else if (phase == 3 && elapsed >= GROUND_TIME) {
>     timer = now; phase = 0;
>   }
> }
>
> void loop() {
>   updateLeg(leg1, legTimer[0], legPhase[0]);
>   updateLeg(leg2, legTimer[1], legPhase[1]);
>   updateLeg(leg3, legTimer[2], legPhase[2]);
>   updateLeg(leg4, legTimer[3], legPhase[3]);
>
>   // 라이더 상체 흔들림 (사인파 근사)
>   static unsigned long riderTimer = 0;
>   if (millis() - riderTimer >= 20) {
>     riderTimer = millis();
>     float phase = (float)(millis() % 1000) / 1000.0; // millis() 직접 float 변환 → 정밀도 손실 방지
>     float angle = 90.0 + 20.0 * sin(2 * PI * phase);
>     rider.write((int)angle);
>   }
> }
> ```

## 서보 범위 보정

서보마다 실제 범위가 다를 수 있음.

```cpp
// 기본: 544~2400 마이크로초
myServo.attach(9);

// 보정: 서보 실제 범위에 맞게 조정
myServo.attach(9, 500, 2500);
// (핀, 최소 펄스 us, 최대 펄스 us)

// 세밀한 제어
myServo.writeMicroseconds(1500); // 직접 펄스폭 지정 (us)
```

## 부드러운 이동 (Sweep)

갑작스러운 각도 변화 → 충격 발생. 단계적 이동으로 부드럽게.

```cpp
void sweepTo(Servo &servo, int fromAngle, int toAngle, int stepDelay) {
  int step = (toAngle > fromAngle) ? 1 : -1;
  for (int angle = fromAngle; angle != toAngle; angle += step) {
    servo.write(angle);
    delay(stepDelay); // stepDelay가 작을수록 빠름 (보통 5~20ms)
  }
  servo.write(toAngle);
}
```

> [!warning] delay() 사용에 주의
> 단일 서보에는 괜찮지만, 다중 서보 동시 제어 시 `millis()` 방식 필요.

## 연속 회전 서보

```cpp
// 연속 회전 서보: 위치가 아닌 속도/방향 제어
myServo.write(90);  // 정지
myServo.write(0);   // 최고속 역방향
myServo.write(180); // 최고속 정방향
// 0~89: 역방향 (빠를수록 작은 값)
// 91~180: 정방향 (빠를수록 큰 값)
```

## PCA9685로 다중 서보 (16채널)

```cpp
#include <Wire.h>
#include <Adafruit_PWMServoDriver.h>

Adafruit_PWMServoDriver pwm = Adafruit_PWMServoDriver(0x40);

#define SERVOMIN  150  // 0도에 해당하는 펄스폭 카운트
#define SERVOMAX  600  // 180도에 해당하는 펄스폭 카운트

void setup() {
  pwm.begin();
  pwm.setPWMFreq(50);  // 서보용 50Hz
}

void setAngle(uint8_t channel, uint8_t angle) {
  uint16_t pulse = map(angle, 0, 180, SERVOMIN, SERVOMAX);
  pwm.setPWM(channel, 0, pulse);
}

void loop() {
  setAngle(0, 0);   // 채널 0, 0도
  setAngle(1, 45);  // 채널 1, 45도
  // ...
}
```
