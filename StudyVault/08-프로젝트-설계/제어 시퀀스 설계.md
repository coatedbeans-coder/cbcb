# 제어 시퀀스 설계 (★★★)

#project #arduino #control

## 시스템 상태 기계 (State Machine)

```
전원 ON
   │
   ▼
[초기화] → 모든 서보 기본 위치
   │
   ▼
[대기] ──→ 버튼 누름 ──→ [갤럽 시작]
   ↑                         │
   │       버튼 다시 누름       ▼
   └───────────────────── [갤럽 실행]
                             │
                         [정지 요청]
                             │
                             ▼
                         [서서히 감속]
                             │
                             ▼
                         [대기 위치로 복귀]
```

## 메인 Arduino 코드 구조

```cpp
#include <Wire.h>
#include <Adafruit_PWMServoDriver.h>

// =============================================
// 설정 상수
// =============================================
#define SERVO_FREQ    50
#define SERVOMIN     150
#define SERVOMAX     600

const int NUM_LEGS   = 4;
const int RIDER_CH   = 4;
const int HEAD_CH    = 5;

// 갤럽 파라미터 (ms)
const int GALLOP_PERIOD = 1000;  // 갤럽 1주기
const int LIFT_DURATION = 200;   // 다리 들기 시간
const int PHASE_OFFSET[4] = {0, 250, 500, 750};

// 각도 설정
const int DOWN_ANGLE = 0;    // 착지 각도
const int UP_ANGLE   = 40;   // 올림 각도

// =============================================
// 전역 변수
// =============================================
Adafruit_PWMServoDriver pwm = Adafruit_PWMServoDriver();

unsigned long gallopStart = 0; // 갤럽 시작 기준 시간 (위상 오프셋 계산용)
int legState[4];  // 0=착지, 1=올림
volatile bool isRunning = false;
int gallopSpeed = 100; // 100% = GALLOP_PERIOD

const int BUTTON_PIN = 2;
const int POT_PIN    = A0;  // 속도 조절 포텐시오미터

// =============================================
// 초기화
// =============================================
void setup() {
  Serial.begin(9600);

  // 서보 드라이버 초기화
  pwm.begin();
  pwm.setOscillatorFrequency(27000000);
  pwm.setPWMFreq(SERVO_FREQ);

  // 버튼 핀
  pinMode(BUTTON_PIN, INPUT_PULLUP);
  attachInterrupt(digitalPinToInterrupt(BUTTON_PIN),
                  onButtonPress, FALLING);

  // 초기 위치 설정
  for (int i = 0; i < NUM_LEGS; i++) {
    setAngle(i, DOWN_ANGLE);
    legState[i] = 0;
  }

  Serial.println("말을 탄 여인 — 준비 완료");
}

// =============================================
// 서보 각도 설정
// =============================================
void setAngle(uint8_t ch, uint8_t angle) {
  uint16_t pulse = map(angle, 0, 180, SERVOMIN, SERVOMAX);
  pwm.setPWM(ch, 0, pulse);
}

// =============================================
// 갤럽 업데이트 (논블로킹)
// =============================================
void updateGallop() {
  if (!isRunning) return;

  unsigned long now = millis();
  // 속도 조절: 포텐시오미터로 주기 500~2000ms
  int period = map(analogRead(POT_PIN), 0, 1023, 500, 2000);

  for (int i = 0; i < NUM_LEGS; i++) {
    // 위상 오프셋 적용: gallopStart 기준 경과 시간 + 각 다리 오프셋
    // unsigned long 덧셈·나머지 연산 → 음수 없음, 오버플로우 안전
    unsigned long phase = (now - gallopStart + (unsigned long)PHASE_OFFSET[i])
                          % (unsigned long)period;

    if (phase < (unsigned long)LIFT_DURATION) {
      if (legState[i] != 1) {
        setAngle(i, UP_ANGLE);
        legState[i] = 1;
      }
    } else {
      if (legState[i] != 0) {
        setAngle(i, DOWN_ANGLE);
        legState[i] = 0;
      }
    }
  }
}

// =============================================
// 라이더 상체 흔들림 (사인파)
// =============================================
void updateRider() {
  static unsigned long riderTimer = 0;
  if (millis() - riderTimer >= 20) {   // 20ms마다 업데이트
    riderTimer = millis();
    // 전후 흔들림: 1Hz 갤럽 동기, ±10°
    // % 연산으로 정규화 → float 정밀도 손실 방지 (장시간 구동 안전)
    float phase = (float)(millis() % 1000) / 1000.0;
    float angle = 90.0 + 10.0 * sin(2 * PI * phase);
    setAngle(RIDER_CH, (int)angle);
  }
}

// =============================================
// 버튼 인터럽트
// =============================================
volatile bool buttonFlag = false;
void onButtonPress() {
  buttonFlag = true;
}

void handleButton() {
  if (!buttonFlag) return;
  buttonFlag = false;

  isRunning = !isRunning;

  if (isRunning) {
    gallopStart = millis();  // 위상 오프셋 기준 시간 기록
    for (int i = 0; i < NUM_LEGS; i++) {
      legState[i] = 0;
    }
    Serial.println("갤럽 시작");
  } else {
    // 모두 착지
    for (int i = 0; i < NUM_LEGS; i++) {
      setAngle(i, DOWN_ANGLE);
    }
    Serial.println("정지");
  }
}

// =============================================
// 메인 루프
// =============================================
void loop() {
  handleButton();
  updateGallop();
  updateRider();
}
```

## 갤럽 속도 조절

```
포텐시오미터 (A0핀):
  0 (최소) → 주기 2000ms → 느린 갤럽 (30 RPM)
  512 (중간) → 주기 1000ms → 갤럽 1 Hz (60 RPM)
  1023 (최대) → 주기 500ms → 빠른 갤럽 (120 RPM)
```

// loop()에 추가
void handleSerial() {
  if (Serial.available()) {
    char cmd = Serial.read();
    switch (cmd) {
      case 's': isRunning = true;  break;  // 시작
      case 'x': isRunning = false; break;  // 정지
      case '+': // 속도 증가
        // 포텐시오미터 대신 소프트웨어 조절
        break;
    }
  }
}
